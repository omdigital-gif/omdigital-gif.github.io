Задача клиента

Интернет-магазин бытовой техники вручную отслеживал цены конкурентов на WB и Ozon — менеджеры тратили 3 часа в день на ручной мониторинг 500 позиций. Задача: автоматизировать сбор данных, выгружать в Excel с подсветкой изменений и запускать по расписанию.

Код — основной модуль парсера

# parser.py — парсер цен WB + Ozon
import requests
import openpyxl
from openpyxl.styles import PatternFill, Font
import time, json
from datetime import datetime
from pathlib import Path

HEADERS = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"}
WB_PRICE_URL = "https://card.wb.ru/cards/v1/detail?appType=1&curr=rub&nm={}"
OZON_PRICE_URL = "https://api.ozon.ru/v2/product/info?product_id={}"

def get_wb_price(article: str) -> dict:
    """Получить цену товара с Wildberries по артикулу"""
    try:
        r = requests.get(WB_PRICE_URL.format(article),
                         headers=HEADERS, timeout=10)
        data = r.json()
        product = data["data"]["products"][0]
        price = product["salePriceU"] / 100
        return {"price": price, "name": product["name"], "source": "WB"}
    except Exception as e:
        return {"price": None, "error": str(e)}

def load_products(path: str) -> list:
    """Загрузить список артикулов из Excel"""
    wb = openpyxl.load_workbook(path)
    ws = wb.active
    products = []
    for row in ws.iter_rows(min_row=2, values_only=True):
        if row[0]:
            products.append({
                "article": row[0],
                "name": row[1],
                "source": row[2],
                "prev_price": row[3] or 0
            })
    return products

def save_report(results: list):
    """Сохранить отчёт в Excel с подсветкой изменений"""
    today = datetime.now().strftime("%Y-%m-%d_%H-%M")
    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = "Цены"

    headers = ["Артикул", "Название", "Источник",
               "Цена сейчас", "Цена прошлая", "Изменение"]
    for col, header in enumerate(headers, 1):
        cell = ws.cell(row=1, column=col, value=header)
        cell.font = Font(bold=True, color="FFFFFF")
        cell.fill = PatternFill("solid", fgColor="1F3A6E")

    red = PatternFill("solid", fgColor="FFD7D7")
    green = PatternFill("solid", fgColor="D7FFDF")

    for i, item in enumerate(results, 2):
        diff = item["price"] - item["prev_price"] if item["prev_price"] else 0
        pct = round(diff / item["prev_price"] * 100, 1) if item["prev_price"] else 0
        row_data = [item["article"], item["name"], item["source"],
                    item["price"], item["prev_price"], f"{pct:+.1f}%"]
        fill = red if pct > 5 else (green if pct < -5 else None)
        for col, val in enumerate(row_data, 1):
            cell = ws.cell(row=i, column=col, value=val)
            if fill: cell.fill = fill

    wb.save(f"prices_{today}.xlsx")
    print(f"Сохранено: prices_{today}.xlsx — {len(results)} позиций")

if __name__ == "__main__":
    products = load_products("products.xlsx")
    results = []
    for p in products:
        data = get_wb_price(p["article"])
        results.append({**p, **data})
        time.sleep(0.3)  # антибан-пауза
    save_report(results)
